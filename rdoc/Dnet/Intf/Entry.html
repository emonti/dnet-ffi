def aliases
  ary = []
  asz = ::Dnet::Addr.size
  p = (self.to_ptr() + Entry.size)  # start at end of struct
  self[:alias_num].times do 
    ary << ::Dnet::Addr.new(p)
    p += asz  
  end
  return ary
end

def copy
  xtra = (::Dnet::Addr.size * self.alias_num)
  if self.len == (Entry.size + xtra)
    super(xtra)
  else
    super()
  end
end
