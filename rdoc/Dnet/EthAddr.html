def addr=(val)
  unless val.to_s =~ /^#{Dnet::Util::RX_MAC_ADDR}$/
    raise(ArgumentError, "invalid mac address #{val.inspect}")
  end
  raw = ::Dnet::Util.unhexify(val, /[:-]/)
  self[:data].to_ptr.write_string(raw, ETH_ADDR_LEN)
end
