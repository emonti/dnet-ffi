def initialize
  if (@handle = ::Dnet.route_open()).address == 0
    raise H_ERR.new("unable to open route handle")
  end
  _handle_opened!
end

def delete(dst)
  _check_open!
  re = Entry.new
  if( re.dst.set_string(dst) and ::Dnet.route_get(@handle, re) == 0 )
    return re 
  end
end
