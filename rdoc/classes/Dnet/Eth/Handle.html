      def eth_send(sbuf)
        _check_open!
        sbuf = sbuf.to_s
        (buf = ::FFI::MemoryPointer.from_string(sbuf)).autorelease = true
        Dnet.eth_send(@handle, buf, sbuf.size)

macaddr()

Retrieves the hardware MAC address for this interface and returns it as a EthAddr object. Uses dnet(3)’s eth_get() function under the hood.

[Source]

# File lib/dnet/eth.rb, line 70
      def macaddr()
        _check_open!
        ea = EthAddr.new
        return nil unless Dnet.eth_get(@handle, ea) == 0
        return ea
      end
macaddr=(ea_str)

Configures the hardware MAC address for this interface to the specified address supplied as a string of colon-separated hex-bytes (example: "de:ad:be:ef:01"). Uses dnet(3)’s addr_aton() to parse ea_str and eth_set() to set the address under the hood.
